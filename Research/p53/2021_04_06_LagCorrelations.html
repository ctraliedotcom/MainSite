def get_correlation(filestr, mode='full', Win=0, do_plot=False):
    """
    Compute cross-correlation between p53 and p21 for a particular 
    Gray level
    
    Parameters
    ----------
    filestr: string
        Path to gray time series
    mode: string
        What mode to do cross-correlation in (full/valid/same)
    Win: int
        Window to use for detrending
    do_plot: boolean
        Whether to save a plot for each individual cross-correlation to disk
    
    Returns
    -------
    lags: ndarray(N)
        Cross-correlation lags
    avg_correlation: ndarray(N)
        Average cross-correlation across all pairs of p53/p21 time series
    peaks: ndarray(N)
        An array of lags, restricted from the range between 0 and 30, that
        led to peaks in each cross-correlation
    """
    p53 = np.loadtxt(filestr.format("p53"))
    p21 = np.loadtxt(filestr.format("p21"))
    divs = np.loadtxt(filestr.format("div"))
    divs = np.sum(divs, 1)
    allndivs = []
    lags = signal.correlation_lags(p53.shape[1], p21.shape[1], mode)
    N = 0
    all_corr = np.array([])
    if do_plot:
        plt.figure(figsize=(10, 10))
    peaks = []
    for i in range(p53.shape[0]): # Loop over all time series
        x1 = p53[i, :]
        x2 = p21[i, :]
        if Win > 0:
            x1 = detrend_timeseries(x1, Win)
            x2 = detrend_timeseries(x2, Win)
        if np.sum(x1 == -1) > 0 or np.sum(x2 == -1) > 0:
            # This signal has missing data, so skip it
            continue
        N += 1
        corr = signal.correlate(x1, x2, mode, 'direct')
        peak = np.argmax(corr[(lags > 0)*(lags < 30)])+1
        if all_corr.size == 0:
            all_corr = corr
        else:
            all_corr += corr
        if do_plot:
            plt.clf()
            plt.subplot(311)
            plt.plot(x1)
            plt.title("p53")
            plt.subplot(312)
            plt.plot(x2)
            plt.title("p21")
            plt.subplot(313)
            plt.plot(lags, corr)
            plt.stem([peak], [3])
            plt.savefig("{}.svg".format(i), bbox_inches='tight')
        peaks.append(peak)
        allndivs.append(divs[i])
    return lags, all_corr/N, np.array(peaks), np.array(allndivs)

def plot_peak_bars(peaks, allndivs):
    """
    Plot bar graphs showing, for different numbers of divisions, how many
    of the time series had a peak cross correlation at 5 or other
    """
    allndivs = np.array(allndivs, dtype=int)
    divs = np.sort(np.unique(allndivs))
    fives = np.zeros(divs.size)
    others = np.zeros(divs.size)
    for peak, div in zip(peaks, allndivs):
        if np.abs(peak-5) <= 1:
            fives[div] += 1
        else:
            others[div] += 1
    ax = plt.gca()
    width = 0.35 # Width of the bars
    rects1 = ax.bar(divs-width/2, fives, width, label="Lag = 4,5,6")
    rects2 = ax.bar(divs+width/2, others, width, label="Lag = Other")
    ax.set_xticks(divs)
    ax.legend()

for Gy in [0, 2, 4, 10]:
    lags, all_corr, peaks, allndivs = get_correlation("data/{}_%iGy.txt"%Gy)
    plt.figure(figsize=(16, 5))
    plt.subplot(131)
    plt.plot(lags, all_corr)
    plt.xlabel("Lag of p21 (in intervals of 30 minutes)")
    plt.ylabel("Cross-Correlation")
    plt.xlim([0, 30])
    plt.title("{} Gy Avg Cross-Correlation".format(Gy))
    plt.subplot(132)
    plt.hist(peaks[peaks < 30])
    plt.title("{} Gy Distribution of peaks".format(Gy))
    plt.xlabel("Lag of p21 (in intervals of 30 minutes)")
    plt.ylabel("Counts")
    plt.subplot(133)
    plot_peak_bars(peaks, allndivs)
    plt.show()
